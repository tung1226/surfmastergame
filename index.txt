/* ============================================================
   DRAW STAR — pure flat colours
============================================================ */
function drawStar(x,y,r,angle){
  var inner=r*0.42, pts=5;
  // outer star dark
  sf('#cc8800');
  ctx.beginPath();
  for(var i=0;i<pts*2;i++){
    var a=(i*Math.PI/pts)-Math.PI/2+angle;
    var rd=(i%2===0)?r*1.08:inner*0.9;
    var px2=x+Math.cos(a)*rd, py2=y+Math.sin(a)*rd;
    i===0?ctx.moveTo(px2,py2):ctx.lineTo(px2,py2);
  }
  ctx.closePath(); ctx.fill();
  // inner star bright
  sf('#ffd700');
  ctx.beginPath();
  for(var i=0;i<pts*2;i++){
    var a=(i*Math.PI/pts)-Math.PI/2+angle;
    var rd=(i%2===0)?r:inner;
    var px2=x+Math.cos(a)*rd, py2=y+Math.sin(a)*rd;
    i===0?ctx.moveTo(px2,py2):ctx.lineTo(px2,py2);
  }
  ctx.closePath(); ctx.fill();
  // highlight
  sf('#fff9a0');
  ctx.beginPath();
  for(var i=0;i<pts*2;i++){
    var a=(i*Math.PI/pts)-Math.PI/2+angle;
    var rd=(i%2===0)?r*0.6:inner*0.55;
    var px2=x+Math.cos(a)*rd, py2=y+Math.sin(a)*rd;
    i===0?ctx.moveTo(px2,py2):ctx.lineTo(px2,py2);
  }
  ctx.closePath(); ctx.fill();
  // stroke
  ss('#ffaa00',Math.max(1.5,r*0.07));
  ctx.beginPath();
  for(var i=0;i<pts*2;i++){
    var a=(i*Math.PI/pts)-Math.PI/2+angle;
    var rd=(i%2===0)?r:inner;
    var px2=x+Math.cos(a)*rd, py2=y+Math.sin(a)*rd;
    i===0?ctx.moveTo(px2,py2):ctx.lineTo(px2,py2);
  }
  ctx.closePath(); ctx.stroke();
}

  sf('#fff4a0'); ctx.beginPath(); ctx.arc(W*0.74,H*0.13,38,0,Math.PI*2); ctx.fill();
  sf('#ffd700'); ctx.beginPath(); ctx.arc(W*0.74,H*0.13,30,0,Math.PI*2); ctx.fill();

  ctx.lineTo(W,H); ctx.closePath(); ctx.fill();
